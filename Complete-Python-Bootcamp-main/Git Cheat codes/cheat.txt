#For adding or deleting the workflow # 

git clone https://github.com/kamaleshsonu/Project-.git    (example)
        ↓
git status
        ↓
Edit/Create files
        ↓
git add .
        ↓
git commit -m "Meaningful message"
        ↓
git push origin main 


#For creating a brand-new repository locally# 
mkdir project
cd project
git init
git add .
git commit -m "Initial commit"
git remote add origin  https://github.com/kamaleshsonu/Project-.git    (example)
git branch -M main
git push origin main
